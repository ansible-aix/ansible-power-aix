> NIM_VIOSUPGRADE    (/home/travis/.ansible/plugins/modules/nim_viosupgrade.py)

        Tool to upgrade VIOSes in NIM environment.

  * This module is maintained by The Ansible Community
OPTIONS (= is mandatory):

= action
        Specifies the operation to perform.
        `altdisk_install' to perform and alternate disk install.
        `alt_disk_clean' to cleanup an existing alternate disk
        install.
        `get_status' to get the status of the upgrade.
        (Choices: altdisk_install, bos_install, get_status)
        type: str

        Backup file name.
        [Default: (null)]
        type: dict

        [Default: (null)]
        type: dict

        [Default: (null)]
        type: dict

        [Default: (null)]
        type: dict

EXAMPLES:

  nim_viosupgrade:
    targets: nimvios01
    action: altdisk_install
